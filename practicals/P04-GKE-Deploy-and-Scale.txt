================================================================================
 PRACTICAL 4 — GKE Cluster: Deploy & Scale an Application
================================================================================
Syllabus : Setting Up Kubernetes Cluster and Deploying & Scaling an App on Kubernetes
Duration : ~2–3 lab hours (cluster create takes time)
Related  : Unit 4 notes; Practical 10 continues monitoring on K8s

--------------------------------------------------------------------------------
1. OBJECTIVE
--------------------------------------------------------------------------------
  • Create a GKE Autopilot (or Standard) cluster
  • Build/push an image to Artifact Registry
  • Deploy with Kubernetes Deployment
  • Scale replicas and expose a LoadBalancer Service
  • Perform a rolling update

--------------------------------------------------------------------------------
2. PREREQUISITES
--------------------------------------------------------------------------------
  • P01–P03 skills helpful
  • Notes: notes/unit-4/CC-II-Unit-IV-Kubernetes-Container-Orchestration.txt
  • Enable APIs: container, artifactregistry, compute

--------------------------------------------------------------------------------
3. ENABLE APIs & VARIABLES
--------------------------------------------------------------------------------
  export PROJECT_ID=$(gcloud config get-value project)
  export REGION=asia-south1
  export CLUSTER=cc2-hello-cluster
  export REPO=cc2-repo
  export APP=hello-app

  gcloud services enable \
    container.googleapis.com \
    artifactregistry.googleapis.com \
    compute.googleapis.com

--------------------------------------------------------------------------------
4. CREATE CLUSTER & CREDENTIALS
--------------------------------------------------------------------------------
  gcloud container clusters create-auto $CLUSTER --region=$REGION

  gcloud container clusters get-credentials $CLUSTER --region=$REGION
  kubectl get nodes
  kubectl cluster-info

--------------------------------------------------------------------------------
5. IMAGE IN ARTIFACT REGISTRY
--------------------------------------------------------------------------------
  gcloud artifacts repositories create $REPO \
    --repository-format=docker --location=$REGION || true

  gcloud auth configure-docker ${REGION}-docker.pkg.dev --quiet

  mkdir -p ~/cc2-gke && cd ~/cc2-gke
  cat > Dockerfile <<'EOF'
  FROM nginx:alpine
  RUN echo "<h1>CC-II Practical 4 — GKE v1</h1>" > /usr/share/nginx/html/index.html
  EXPOSE 80
  EOF

  export IMAGE=${REGION}-docker.pkg.dev/${PROJECT_ID}/${REPO}/${APP}:v1
  docker build -t $IMAGE .
  docker push $IMAGE

--------------------------------------------------------------------------------
6. DEPLOY, SCALE, EXPOSE
--------------------------------------------------------------------------------
  kubectl create deployment $APP --image=$IMAGE
  kubectl scale deployment $APP --replicas=3
  kubectl get pods -o wide

  kubectl expose deployment $APP \
    --name=${APP}-svc \
    --type=LoadBalancer \
    --port=80 \
    --target-port=80

  kubectl get service ${APP}-svc
  # Wait until EXTERNAL-IP is not <pending>
  curl http://EXTERNAL_IP/

--------------------------------------------------------------------------------
7. ROLLING UPDATE TO v2
--------------------------------------------------------------------------------
  # Edit Dockerfile message to "GKE v2", rebuild:
  export IMAGE_V2=${REGION}-docker.pkg.dev/${PROJECT_ID}/${REPO}/${APP}:v2
  docker build -t $IMAGE_V2 .
  docker push $IMAGE_V2

  kubectl set image deployment/$APP ${APP}=$IMAGE_V2
  # If container name differs, check: kubectl get deploy $APP -o yaml | grep -A2 image
  # Autogenerated container name is often same as deployment name for create deployment

  kubectl rollout status deployment/$APP
  kubectl get pods
  curl http://EXTERNAL_IP/

  # Optional rollback demo:
  # kubectl rollout undo deployment/$APP

--------------------------------------------------------------------------------
8. OPTIONAL — HPA
--------------------------------------------------------------------------------
  kubectl autoscale deployment $APP --cpu-percent=80 --min=2 --max=6
  kubectl get hpa

--------------------------------------------------------------------------------
9. VERIFY
--------------------------------------------------------------------------------
  [ ] Nodes Ready
  [ ] 3 (or HPA) Pods Running
  [ ] LoadBalancer EXTERNAL-IP serves HTML
  [ ] Rollout to v2 succeeded

--------------------------------------------------------------------------------
10. DELIVERABLES
--------------------------------------------------------------------------------

  2. Screenshot: kubectl get pods,svc
  3. Screenshot: browser hit on EXTERNAL-IP (v1 and v2 if possible)
  4. Short explanation: Pod vs Deployment vs Service

--------------------------------------------------------------------------------
12. CLEANUP (IMPORTANT — CLUSTERS COST MONEY)
--------------------------------------------------------------------------------
  kubectl delete service ${APP}-svc --ignore-not-found
  kubectl delete deployment $APP --ignore-not-found
  gcloud container clusters delete $CLUSTER --region=$REGION --quiet

  # Keep cluster ONLY if you will continue immediately with Practical 10
